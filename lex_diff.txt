+        if (!hasMiles && !hasPrice) return null;

+        let origin = '', destination = '';

+        );

+            origin = routeMatch[1].trim();

+        // Extraer millas
+        let miles = 0;

+
+        // Extraer precio total
+        let rate = 0;
+        const rateMatch = t.match(/\$\s*(\d[\d,]*(?:\.\d+)?)/);

+
+        // Extraer RPM si viene explícito

+        const rpmMatch = t.match(/(\d+\.\d+)\s*(?:rpm|\/mi|per mi)/i);
+        if (rpmMatch) rpm = parseFloat(rpmMatch[1]);

+        const yearStart = new Date(Date.UTC(tmp.getUTCFullYear(), 0, 1));

+        const weekKey = `${tmp.getUTCFullYear()}-W${weekNum}`;
